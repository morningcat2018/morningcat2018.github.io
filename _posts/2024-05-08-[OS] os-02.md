## 解决异常退出

程序自己是不能 “停下来” 的

只能借助操作系统

```
movq $SYS_exit,  %rax   # exit(
movq $1,         %rdi   #   status=1
syscall                 # );

把 “系统调用” 的参数放到寄存器中
执行 syscall，操作系统接管程序  操作系统可以任意改变程序状态 (甚至终止程序)
```

## 接着探讨最小的程序

```asm
// https://www.huii.top/938.html
#include <sys/syscall.h>

#define syscall3(id, a1, a2, a3) \
    movq $SYS_##id, %rax; \
    movq $a1, %rdi; \
    movq $a2, %rsi; \
    movq $a3, %rdx; \
    syscall

#define syscall1(id, a1) syscall3(id, a1, 0, 0)

.globl _start
_start:
    syscall3(write, 1, addr1, addr2 - addr1)
    syscall1(exit, 1)

addr1:
    .ascii "\033[01;31mHello, world\033[0m\n"
addr2:
```

> gcc -g -S main.s > minimal.s

## 另一种实现

```asm
#include <sys/syscall.h>
 
.globl _start
_start:
  movq $SYS_write, %rax   // write(
  movq $1,         %rdi   //   fd=1,
  movq $st,        %rsi   //   buf=st,
  movq $(ed - st), %rdx   //   count=ed-st
  syscall                 // );
 
  movq $SYS_exit,  %rax   // exit(
  movq $1,         %rdi   //   status=1
  syscall                 // );
 
st:
  .ascii "\033[01;31mHello, OS World\033[0m\n"
ed:
```

## macos下的实践

hello.asm

```asm
section .data
    hello db 'Hello, World!',10,0  ; 注意这里加上换行符和终止符

section .text
    global _start

_start:
    ; write(1, message, 14)
    mov rax, 0x2000004         ; 系统调用号 4 表示 write
    mov rdi, 1                 ; 文件描述符 1 表示标准输出
    mov rsi, hello             ; 要打印的字符串
    mov rdx, 14                ; 字符串的长度
    syscall

    ; exit(0)
    mov rax, 0x2000001         ; 系统调用号 1 表示 exit
    xor edi, edi               ; 返回值设为 0
    syscall
```
